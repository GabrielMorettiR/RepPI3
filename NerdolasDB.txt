create table cliente(
nome char(30),
cpf char(11) primary key,
email char(40),
telefone char(11),
endereco char(40)
);

create table produto(
id int primary key generated always as identity,
nome char(30),
descricao char(60),
idcategoria smallint,
quantidade int,
preco double
);

create table venda(
idvenda int primary key generated always as identity,
id int,
cpf char(11),
quantidade int,
preco double
);

create table categoria(
id smallint primary key generated always as identity,
titulo char(20)
);

create table filial(
idfilial int primary key generated always as identity,
nome char(30),
endereco char(40)
)

create table usuario(
nome varchar(50),
login varchar(50),
senha varchar(50),
perfil varchar(50)
)

ALTER TABLE produto ADD CONSTRAINT idcategoria
FOREIGN KEY(idcategoria) REFERENCES categoria(id);

ALTER TABLE venda ADD CONSTRAINT idproduto
FOREIGN KEY(id) REFERENCES produto(id);

ALTER TABLE venda ADD CONSTRAINT cpf
FOREIGN KEY(cpf) REFERENCES cliente(cpf);

ALTER TABLE venda ADD CONSTRAINT idfilial
FOREIGN KEY(idfilial) REFERENCES filial(idfilial);
